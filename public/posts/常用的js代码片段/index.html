1.复制文本

function copyToClip(content, message){
  let aux = document.createElement("input");
  aux.setAttribute("value", content);
  document.body.appendChild(aux);
  aux.select();
  document.execCommand("copy");
  document.body.removeChild(aux);
  if (!message) {
   alert('复制成功')
  } else {
    alert(message)
  }
}
