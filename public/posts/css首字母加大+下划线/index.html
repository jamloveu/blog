首字母加大+下划线

jam 收录于  draft
 2017-04-15   约 44 字   预计阅读 1 分钟 

<!doctype html>
<html>
	<head>
		<style type="text/css">
			.text{
				font-family:Times New Roman;
				font-size:24px;
				float:left;
				clear:left;
				width:auto;
			}
			.text:first-letter{
				font-size:48px;
			}
			.text:after {
				content:'';
				display:block;
				margin-top:-5px;
				border-bottom:2px solid grey;
			}
		</style>
		<script src="js/jquery-1.8.2.min.js">
			$(document).ready(function(){
				$(.text).each(function(){
					$(this:after).css("width",$(this).css("width"));
				});
			});
		</script>
	</head>
	<body>
		<div class="text">About</div>
		<div class="text">Technology</div>
	</body>
</html>
